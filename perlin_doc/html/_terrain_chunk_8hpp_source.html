164 HVertexStatus vertexValuation(float x, float z){

167 glm::vec3 actualC;
168
169 double oauxHX, oauxHZ;
170 glm::vec3 oauxCX, oauxCZ;

172 r.BioTp = getBiomeXZ(x, z);
173 double auxfxz = (double) this->gridSize/bvbuild[r.BioTp].bFrequence;//bvbuild[r.BioTp]

175 r.BaseH = pNoise.octaveNoise((double)x/auxfxz, (double)z/auxfxz, bvbuild[r.BioTp].bOctaves);
176
177 actualC = biomeColorValuation(r.BioTp, r.BaseH);
178
179 actualH = biomeHeightValuation(r.BioTp, r.BaseH);
180
181 BorderHelper someAuxforX = borderTest(x), someAuxforZ = borderTest(z);
182
183
184
185 BiomeType otherSide_x, otherSide_z, otherSide_xz;
186 double beyondTheWallBase_x, beyondTheWallFinal_x;
187 double beyondTheWallBase_z, beyondTheWallFinal_z;
188 double beyondTheWallBase_xz, beyondTheWallFinal_xz;
189
190
191 r.FinalH = actualH;
192 r.ColorH = actualC;
193
194 if(someAuxforX.isBorder){
195 if(someAuxforX.borderAbove){
196 // The actual point(x, z), is border in the x axis
197 otherSide_x = getBiomeXZ(x + this->biomeBorderLen, z);
198 auxfxz = (double) this->gridSize/bvbuild[otherSide_x].bFrequence;
199 beyondTheWallBase_x = pNoise.octaveNoise((double)x/auxfxz, (double)z/auxfxz, bvbuild[otherSide_x].bOctaves);
200 beyondTheWallFinal_x = biomeHeightValuation(otherSide_x, beyondTheWallBase_x);
201 oauxHX = glm::mix(beyondTheWallFinal_x, actualH, someAuxforX.pDistToBorder);
202 r.FinalH = oauxHX;
203 oauxCX = glm::mix(biomeColorValuation(otherSide_x, beyondTheWallBase_x), actualC, someAuxforX.pDistToBorder);
204 r.ColorH = oauxCX;
205 }
206
207 }
208
209 if(someAuxforZ.isBorder){
210 if(someAuxforZ.borderAbove){
211 // The actual point(x, z), is border in the z axis
212 otherSide_z = getBiomeXZ(x, z + this->biomeBorderLen);
213 auxfxz = (double) this->gridSize/bvbuild[otherSide_z].bFrequence;
214 beyondTheWallBase_z = pNoise.octaveNoise((double)x/auxfxz, (double)z/auxfxz, bvbuild[otherSide_z].bOctaves);
215 beyondTheWallFinal_z = biomeHeightValuation(otherSide_z, beyondTheWallBase_z);
216 oauxHZ = glm::mix(beyondTheWallFinal_z, actualH, someAuxforZ.pDistToBorder);
217 r.FinalH = oauxHZ;
218 oauxCZ = glm::mix(biomeColorValuation(otherSide_z, beyondTheWallBase_z), actualC, someAuxforZ.pDistToBorder);
219 r.ColorH = oauxCZ;
220 }
221
222 }
223
224 if(someAuxforX.isBorder && someAuxforZ.isBorder){
225 if( someAuxforX.borderAbove && someAuxforZ.borderAbove){
226 // The actual point(x, z), is border in the x and z axes
227 otherSide_xz = getBiomeXZ(x + this->biomeBorderLen, z + this->biomeBorderLen);
228 auxfxz = (double) this->gridSize/bvbuild[otherSide_xz].bFrequence;

232 //Bilinear interpolation

234 r.FinalH = glm::mix(beyondTheWallFinal_xz, beyondTheWallFinal_x, someAuxforZ.pDistToBorder);

252 BorderHelper borderTest(float k){
253 bool negativeFlag;
254 long int kaux;
255 negativeFlag = k < 0.0 ? true : false;
256 kaux = negativeFlag? k*-1: k;
257 long int nearborder = ((kaux / this->biomeSize) + 1) * this->biomeSize;
258 long int nearborder2 = ((kaux / this->biomeSize)) * this->biomeSize;
259
260 if(negativeFlag){
261 nearborder *= -1;
262 nearborder2 *= -1;
263 }
264
265 BorderHelper borderRes;
266 if(glm::abs(nearborder - k) < this->biomeBorderLen){
267 borderRes.isBorder = true;
268 borderRes.borderAbove = true;
269 borderRes.pDistToBorder = glm::abs(nearborder - k) / this->biomeBorderLen;
270 }else if(glm::abs(nearborder2 - k) < this->biomeBorderLen){
271 borderRes.isBorder = true;
272 borderRes.borderAbove = false;
273 borderRes.pDistToBorder = glm::abs(nearborder2 - k) / this->biomeBorderLen;
274 }else{
275 borderRes.isBorder = false;
276 }
277
278 if(negativeFlag){
279 borderRes.borderAbove = !borderRes.borderAbove;
280 }
281 return borderRes;
282 }

290 BiomeType getBiomeXZ(double px, double pz){
291 double dxs = (px/(double)this->biomeSize);
292 double dzs = (pz/(double)this->biomeSize);
293 if(dxs < 0) dxs -=1.0;
294 if(dzs < 0) dzs -=1.0;
295
296 dxs = glm::trunc(dxs);
297 dzs = glm::trunc(dzs);
298
299 /* to force four different adjacent biomes (worst case)
300 int di, dj;
301 di = dxs;
302 dj = dzs;
303
304 if(di == 0 && dj == 0){
305 return DESERT;
306 }else if(di == 0 && dj == 1){
307 return MOUNTAINS;
308 }else if(di == 1 && dj == 0){
309 return VALLEY;
310 }else if(di == 1 && dj == 1){
311 return PLAINS;
312 }
313 */
314
315 double areaValue = pNoise.noise0_1((dxs + 7.3)/0.5, (dzs+ 7.3)/0.5);
316 areaValue = glm::clamp(areaValue, 0.0, 1.0);
317
318
319 BiomeType toTest =(BiomeType) (areaValue * BIOMESN);
320
321
322 return toTest;
323 }
